import numpy as np

def check_vector_stability(vector: np.ndarray) -> dict:
    """
    Check numerical stability of input vector and return a diagnostic report.
    
    Checks for:
    1. NaN/Inf values
    2. Extreme values (>1e12 in magnitude)
    3. Value distribution variance
    4. Potential overflow risks in activation functions
    
    Returns:
        Dictionary containing stability metrics and warnings
    """
    # Initialize report
    report = {
        'stability_score': 1.0,
        'warnings': [],
        'checks': {}
    }

    # 1. Check for NaN/Inf values
    nan_inf_count = np.isnan(vector).sum() + np.isinf(vector).sum()
    if nan_inf_count > 0:
        report['stability_score'] -= 0.3
        report['warnings'].append(
            f"NaN or Infinite values detected. Clip to inf to -inf and NaN to 0.")
        report['checks']['nan_inf_check'] = {
            'count': nan_inf_count,
            'values': []
        }

    # 2. Check for extreme values (>1e12 magnitude)
    extreme_mask = np.abs(vector) > 1e12
    extreme_count = np.sum(extreme_mask)
    if extreme_count > 0:
        report['stability_score'] -= 0.3
        report['warnings'].append(
            f"Extreme values (magnitude >1e12) detected. Consider scaling down inputs.\nExamples: {vector[extreme_mask][:5]}")
        report['checks']['extreme_values_check'] = {
            'count': extreme_count,
            'indices': np.where(extreme_mask)[0][:5],
            'values': vector[extreme_mask][:5].tolist()
        }

    # 3. Check value distribution variance
    std_dev = np.std(vector)
    if std_dev > 1e10:
        report['stability_score'] -= 0.3
        report['warnings'].append(
            f"High value variance in vector. Consider normalizing data. (std={std_dev:.2e})")
        report['checks']['variance_check'] = {
            'std_dev': std_dev,
            'mean': np.mean(vector)
        }

    # 4. Check potential overflow in activation functions
    # Sigmoid overflow check
    if extreme_mask.any():  # Already checked above, but repeat for emphasis
        report['stability_score'] -= 0.1
        report['warnings'].append(
            "Extreme values in input vector could cause sigmoid saturation. Check activation function inputs.")

    # Calculate final score (minimum 0.1)
    report['stability_score'] = max(0.1, report['stability_score'])

    return report
