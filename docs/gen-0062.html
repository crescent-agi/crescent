```python
import json, os, re

def extract_gen_num(path):
    # Extract generation number from path like artifacts/agi_core_continuous_trained_gen42_20ep/training_stats.json
    m = re.search(r'_gen(\d+)', path)
    return int(m.group(1)) if m else None

def collect_stats(root_dir='artifacts'):
    rows = []
    for dirpath, _, filenames in os.walk(root_dir):
        if 'training_stats.json' in filenames:
            path = os.path.join(dirpath, 'training_stats.json')
            try:
                with open(path, 'r') as f:
                    data = json.load(f)
                # Flatten: take known keys
                row = {
                    'path': path,
                    'gen': extract_gen_num(path),
                    'stats': data
                }
                rows.append(row)
            except Exception as e:
                print(f"Failed to read {path}: {e}")
    return rows

def main():
    rows = collect_stats()
    # Sort by generation descending if available
    rows.sort(key=lambda r: r['gen'] if r['gen'] is not None else 0, reverse=True)
    # Print a simple table
    print(f"{'gen':<4} {'path':<60} {'keys':<20}")
    for r in rows:
        keys = ','.join(r['stats'].keys())
        print(f"{r['gen']:<4} {r['path'][:60]:<60} {keys:<20}")

if __name__ == '__main__':
    main()
```
